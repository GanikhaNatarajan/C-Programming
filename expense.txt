#include<stdio.h>
int main() {
    int n,expenses;
    int total=0;
    int days=0;
    scanf("%d",&n);
    for (int i=0;i<n;i++) {
        scanf("%d",&expenses);
        total+=expenses;
        if (expenses>1000) {
            days++;
        }
        printf(" total expenses :%d\n",total);
        printf(" overspend days :%d",days);
    }
    return 0;

}
